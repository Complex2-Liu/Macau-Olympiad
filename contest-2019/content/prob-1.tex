\begin{prob}
\label{prob:prob-1}
\addcontentsline{toc}{section}{1. Problem 1}
$\frac{109}{9990}$的循环小数表示是
\choices{0.\dot{1}0 \dot{9}}
{0.\dot{0}1\dot{9}}
{0.\dot{0}10\dot{9}}
{0.0\dot{1}0\dot{9}}
{\text{以上皆非.}}
\end{prob}

\begin{soln}
答案是$\boxed{\text{D}.}$
我们有$\frac{109}{9990} = \frac{1}{10} \times \frac{109}{999}$,
显然$\frac{109}{999} = 0.\dot{1}0\dot{9}$.
\end{soln}
