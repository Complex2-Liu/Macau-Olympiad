\begin{prob}
\label{prob:prob-12}
\addcontentsline{toc}{section}{12. Problem 12}
\begin{enumerate}[label={\normalfont(\alph*)}]
\item 设 $f(x) = x^3 - 3abx + a^3 + b^3$, 化简 $f(-a-b)$.
\item 因式分解 $a^3 + b^3 + c^3 - 3abc$.
\end{enumerate}
\end{prob}

\begin{soln}
(a) 答案是 $\boxed{0.}$

\bigskip

(b) 熟知 $\boxed{a^3 + b^3 + c^3 - 3abc = (a+b+c)(a^2+b^2+c^2-ab-bc-ca).}$
\end{soln}
