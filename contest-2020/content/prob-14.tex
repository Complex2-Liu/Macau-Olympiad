\begin{prob}
\label{prob:prob-14}
\addcontentsline{toc}{section}{14. Problem 14}
试求满足 $ab^3 = -135$ 及 $(a+b)b = -6$ 的所有实数对 $(a,b)$.
\end{prob}

\begin{soln}
我们有 $ab = -\frac{135}{b^2} \implies -6 = b^2 + ab =
b^2 - \frac{135}{b^2} \implies u^2 + 6u - 135 = 0, u = b^2 \ne 0$.
解得 $u = -15, 9$, 答案是 $\boxed{(a,b) = (-5,3), (5,-3).}$
\end{soln}
