\begin{prob}
\label{prob:prob-13}
\addcontentsline{toc}{section}{13. Problem 13}
把 $2020$ 写成五个 (不一定互异) 整数的立方之和.
\end{prob}

\begin{soln}
$\boxed{2020 = 10^3 + 10^3 + 3^3 - 2^3 + 1^3.}$
\end{soln}
