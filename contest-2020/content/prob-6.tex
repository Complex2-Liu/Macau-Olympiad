\begin{prob}
\label{prob:prob-6}
\addcontentsline{toc}{section}{6. Problem 6}
$N = 2020^2 - 2019^2 + 2018^2 - \cdots + 2^2 - 1^2.$
\choices{N<0}{N=0}{0<N<2020}{2020\le N < 4040}{\text{以上皆非.}}
\end{prob}

\begin{soln}
$N > 2020^2 - 2019^2 + 2^2 - 1^2 = 4039 + 3 > 4040$.
答案是 $\boxed{\textbf{E}.}$
\end{soln}
