\begin{prob}
\label{prob:prob-2}
\addcontentsline{toc}{section}{2. Problem 2}
记 $P(x) = x^2 + ax + b$, 其中 $a,b \in [-2,2]$.
当 $a,b$ 在闭区间 $[-2,2]$ 上变动时,
求二次方程 $P(x) = 0$ 的实数解的取值范围.
\end{prob}

\begin{soln}
$\Delta = a^2 - 4b \le 12$. 易知答案为
\[
\boxed{\frac{-2 - \sqrt{12}}{2}} \le
\frac{-a \pm \sqrt{\Delta}}{2} \le
\boxed{\frac{2 + \sqrt{12}}{2}.}
\]
\end{soln}
